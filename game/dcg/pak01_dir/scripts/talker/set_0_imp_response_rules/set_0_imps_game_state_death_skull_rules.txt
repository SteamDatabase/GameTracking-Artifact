// GAINED death skull ================================================================================================
//====================================================================================================================

// there are a bunch of these rules as we need to check for death skulls from all the concepts that can trigger the state
// these have the "ThisIsImportant" criteria to basically force the response to play if any less important things like damage etc happen and also trigger a valid response
// using "ThisIsImportant" saves having to tune weighting on individual criterion
// ApplyContext "IsWaitingToPlayPlayerTowerDeathSkullResponses:1:15" forces the response to wait 15 seconds even if its triggered again, to stop the animation happening too often in a small time window which is possible.

// player tower: ***************************************************************************************************************************************************************************************************************************************************************************************************************************


Rule rule_imps_local_negative_player_tower_gained_death_skull_summon
{
	criteria Summon IsNotCombat LocalTowerGainedDeathSkull IsLocalImp IsWaitingToPlayPlayerTowerDeathSkullResponses ThisIsImportant
	ApplyContext "IsWaitingToPlayPlayerTowerDeathSkullResponses:1:15"
	ApplyContextToWorld
	response imps_local_negative_player_tower_gained_death_skull
}

{
