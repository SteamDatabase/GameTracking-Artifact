
//====   TIMER  BASED RULES ABOUT THE GAME CLOCK ==============================================================================================================================================================================================
//=============================================================================================================================================================================================================================================


// note: these 'NoRepeat' criteria are needed to stop the response triggering with every call of the timer ( otherwise they would retrigger every second )

// players timer:


Rule rule_imps_local_10_secs_remain_from_timer
{
	criteria Timer IsLocalImp IsLoE10SecondsRemaining IsGoE8SecondsRemaining IsActingPlayerLocal IsActionPhase NoRepeatForPlayerTimerBasedClockResponses 
	ApplyContext "NoRepeatForPlayerTimerBasedClockResponses:1:3"
	ApplyContextToWorld	
	ApplyContext "isPlayer10SecondsRemainingCurrentlyActive:1:11"
	ApplyContextToWorld		
	response imps_local_10_secs_remain
}

Rule rule_imps_local_0_secs_remain_from_timer
{
	criteria Timer IsLocalImp IsLoE2SecondsRemaining IsGoE0SecondsRemaining IsActingPlayerLocal IsActionPhase NoRepeatForPlayerTimerBasedClockResponses
	ApplyContext "NoRepeatForPlayerTimerBasedClockResponses:1:3"
	ApplyContextToWorld		
	response imps_local_0_secs_remain
}


// opponents timer:


Rule rule_imps_opponent_10_secs_remain_from_timer
{
	criteria Timer IsOpponentImp IsLoE10SecondsRemaining IsGoE8SecondsRemaining IsNotActingPlayerLocal IsActionPhase NoRepeatForOpponentTimerBasedClockResponses 
	ApplyContext "NoRepeatForOpponentTimerBasedClockResponses:1:3"
	ApplyContextToWorld		
	response imps_opponent_10_secs_remain
}
