Rule rule_imps_local_winning_game_idle_from_timer
{
	// MOMENTUM: this will trigger an 'Im winning' idle on the players imp that will stomp the imps regular idles if the player is winning overall
	
	criteria Timer IsActionPhase IsTimeRemainingInLane IsLocalImp PlayerIsWinningGame IsGameRound4orMore NoLocalTowerDeathSkull NoRepeatForPlayerTimerBasedWinLoseIdleResponses
	ApplyContext "NoRepeatForPlayerTimerBasedWinLoseIdleResponses:1:25"
	ApplyContextToWorld		
	response imps_local_winning_game_idle
}
