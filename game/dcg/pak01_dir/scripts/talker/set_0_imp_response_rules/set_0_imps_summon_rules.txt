

//====   PLAYER SUMMONS         ======================================================================================
//====================================================================================================================


// note: most of the generic responses that happen frequently are timed by NoRepeatForPlayerCommonResponses, but summon is not otherwise it triggers the card_swaps reaction as usually a summon will fire a change of target event 


Rule rule_imps_local_positive_summon
{
	criteria Summon IsNotCombat IsLocalImp LocalChangeMomentum_Pos IsSourceLocal OpponentChangeMomentum_Neg 	
	
	response imps_local_positive_small
	// using town portal goodbye wave as a random variant as it works as greeting too
	response imps_local_play_card_town_portal_positive
	response imps_do_nothing
}

Rule rule_imps_local_positive_summon_improvement
{
	criteria Summon IsNotCombat IsLocalImp LocalChangeMomentum_Pos IsSourceLocal OpponentChangeMomentum_Neg isTargetImprovement	
	
	response imps_local_positive_small_summon_improvement
	response imps_do_nothing
}


Rule rule_imps_opponent_negative_summon
{
	criteria Summon IsNotCombat IsOpponentImp IsSourceLocal LocalChangeMomentum_Pos OpponentChangeMomentum_Neg 		
		
	response imps_opponent_negative_small
	response imps_do_nothing	
}


Rule rule_imps_opponent_negative_summon_multi_creep
{
	criteria Summon IsNotCombat IsOpponentImp IsSourceLocal LocalChangeMomentum_PosMid OpponentChangeMomentum_Neg 
	
	response imps_opponent_negative_broad
	response imps_do_nothing	
}

Rule rule_imps_local_negative_summon_losing_game
{
	
  // MOMENTUM: if the player is BADLY LOSING the overall game, their imp will react negativley to a simple summon 
	
	criteria Summon IsNotCombat IsLocalImp IsSourceNotLocal OpponentChangeMomentum_Pos PlayerIsLosingGame IsGameRound4orMore 	
		
	response imps_local_negative_broad
	response imps_do_nothing	

}
